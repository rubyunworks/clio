    def Size(*t)
      if t.size == 0
        col, row = API.GetConsoleScreenBufferInfo(@handle )
        return [col, row]
      else
        row = -1 if !t[1]
        col = -1 if !t[0]
        if col < 0 or row < 0
          curr_col, curr_row = Size()
          col = curr_col if col < 0
          row = curr_row if row < 0
        end
        API.SetConsoleScreenBufferSize(@handle, row, col)
      end
    end
