    def ReadRect( left, top, right, bottom )
      col = right  - left + 1
      row = bottom - top  + 1
      size = col * row
      buffer = ' ' * size * 4
      if API.ReadConsoleOutput( @handle, buffer, col, row, 0, 0, 
                               left, top, right, bottom )
        #return buffer.unpack('US'*size)  # for unicode
        return buffer.unpack('axS'*size)  # for ascii
      else
        return nil
      end
    end
